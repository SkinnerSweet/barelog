44 #if BARELOG_SAFE_MODE
45 #define barelog_get_mutex(core, mutex) do { \
46  if (manager.read(mem_space[BARELOG_NB_CORES].base + (core), 1, &(mutex)) != BARELOG_SUCCESS) { \
47  return BARELOG_SHRMEM_READ_ERR; \
48  } \
49 } while(0)
50 
51 #define barelog_set_mutex(core, value) do { \
52  uint8_t set_mutex_tmp = (value); \
53  if (manager.write(mem_space[BARELOG_NB_CORES].base + (core), 1, &set_mutex_tmp) != BARELOG_SUCCESS) { \
54  return BARELOG_SHRMEM_WRITE_ERR; \
55  } \
56 } while(0)
57 
58 #define barelog_try_mutex(core) do { \
59 uint8_t __mutex = 1; \
60 uint32_t __timeout = BARELOG_MUTEX_TRY_MAX; \
61 while (__mutex) { \
62  --__timeout; \
63  barelog_get_mutex((core), __mutex); \
64  if (__timeout <= 0) { \
65  return BARELOG_TIMEOUT_ERR; \
66  } \
67 } \
68 } while(0)
69 #else

72 #define barelog_set_mutex(core, value)

78 int8_t host_mem_manager_init(const barelog_platform_t platform,
79  void * (*init)(void *address, size_t size, void *data),
80  int8_t (*read)(const void *address, size_t size, void *buffer),
81  int8_t (*write)(void *address, size_t size, const void *buffer),
82  int8_t (*finalize)(void *mem_space)) {
83 
84 #if BARELOG_CHECK_MODE || BARELOG_DEBUG_MODE
85  if (manager.initialized || !init || !finalize || !platform.mem_space.phy_base) {
86  return BARELOG_UNINITIALIZED_PARAM_ERR;
87  }
88 
89  if (platform.mem_space.length <= BARELOG_SHARED_MEM_MAX) {
90  return BARELOG_INIT_ERR;
91  }
92 #endif
93 
94  manager.init = init;
95  manager.read = read;
96  manager.write = write;
97  manager.finalize = finalize;
98  // TODO : alignment concerns...
99 
100  /* Barelog's configuration areas : */
101  // Memory synchronization (mutexes between host and device) :
102 #if BARELOG_SAFE_MODE
103  mem_space[BARELOG_SAFE_MODE_I].phy_base = platform.mem_space.phy_base;
104  mem_space[BARELOG_SAFE_MODE_I].length = BARELOG_SAFE_MEM_SIZE;
105  mem_space[BARELOG_SAFE_MODE_I].alignment = platform.mem_space.alignment;
106  mem_space[BARELOG_SAFE_MODE_I].word_size = platform.mem_space.word_size;
107  mem_space[BARELOG_SAFE_MODE_I].data = calloc(1, BARELOG_MEM_SPACE_DATA_SIZE);
108  mem_space[BARELOG_SAFE_MODE_I].base = manager.init(mem_space[BARELOG_SAFE_MODE_I].phy_base,
109  mem_space[BARELOG_SAFE_MODE_I].length,
110  mem_space[BARELOG_SAFE_MODE_I].data);
111  if (mem_space[BARELOG_SAFE_MODE_I].base == NULL) {
112  free(mem_space[BARELOG_SAFE_MODE_I].data);
113  return BARELOG_ERR;
114  }
115  memset(mem_space[BARELOG_SAFE_MODE_I].base, 0, BARELOG_NB_MUTEX_BYTES);
116 #endif // BARELOG_SAFE_MODE
117 
118 #if BARELOG_DEBUG_MODE
119  manager.mem_space[BARELOG_DEBUG_MODE_I].phy_base = platform.mem_space.phy_base + BARELOG_DEBUG_OFF;
120  manager.mem_space[BARELOG_DEBUG_MODE_I].length = BARELOG_DEBUG_MEM_SIZE;
121  manager.mem_space[BARELOG_DEBUG_MODE_I].alignment = platform.mem_space.alignment;
122  manager.mem_space[BARELOG_DEBUG_MODE_I].word_size = platform.mem_space.word_size;
123  manager.mem_space[BARELOG_DEBUG_MODE_I].data = calloc(1, BARELOG_MEM_SPACE_DATA_SIZE);
124  manager.mem_space[BARELOG_DEBUG_MODE_I].base = manager.init(manager.mem_space[BARELOG_DEBUG_MODE_I].phy_base,
125  manager.mem_space[BARELOG_DEBUG_MODE_I].length,
126  manager.mem_space[BARELOG_DEBUG_MODE_I].data);
127  if (manager.mem_space[BARELOG_DEBUG_MODE_I].base == NULL) {
128  free(manager.mem_space[BARELOG_DEBUG_MODE_I].data);
129  return BARELOG_ERR;
130  }
131  memset(manager.mem_space[BARELOG_DEBUG_MODE_I].base, 0, BARELOG_DEBUG_MEM_SIZE);
132 #endif // BARELOG_DEBUG_MODE
133  /* End of Barelog's configuration areas. */
134 
135  /* Barelog's data areas, used to store events in shared memory : */
136  void *base = platform.mem_space.phy_base + BARELOG_SHARED_MEM_DATA_OFFSET;
137  for (uint32_t i = 0; i < BARELOG_NB_CORES; ++i) {
138  manager.mem_space[i].phy_base = base
139  + i * BARELOG_SHARED_MEM_PER_CORE_MAX;
140  manager.mem_space[i].length = BARELOG_SHARED_MEM_PER_CORE_MAX;
141  manager.mem_space[i].alignment = platform.mem_space.alignment;
142  manager.mem_space[i].word_size = platform.mem_space.word_size;
143  manager.mem_space[i].data = calloc(1, BARELOG_MEM_SPACE_DATA_SIZE);
144  manager.mem_space[i].base = manager.init(manager.mem_space[i].phy_base,
145  manager.mem_space[i].length, manager.mem_space[i].data);
146  if (manager.mem_space[i].base == NULL) {
147  free(manager.mem_space[i].data);
148  return i - 1;
149  }
150  memset(manager.mem_space[i].base, 0, manager.mem_space[i].length);
151  }
152  /* End of Barelog's data areas. */
153 
154  manager.initialized = 1;
155 
156  return BARELOG_NB_CORES;
157 }

159 int8_t host_mem_manager_finalize(void) {
160 
161 #if BARELOG_CHECK_MODE || BARELOG_DEBUG_MODE
162  if (!manager.initialized) {
163  return BARELOG_INCONSISTENT_PARAM_ERR;
164  }
165 #endif
166 
167  for (uint32_t i = 0; i < BARELOG_HOST_NB_MEM_SPACE; ++i) {
168  if (manager.finalize(manager.mem_space[i].data) != BARELOG_SUCCESS) {
169  /* FIXME : it could not always be the "data" field that corresponds to
170  * the data to deallocate. A safest way to proceed could be
171  * to do a reverse on mmap and to clean the corresponding allocated
172  * area since we know that "data" corresponds to an address created
173  * by the use of mmap.
174  */
175  return i;
176  }
177  free(manager.mem_space[i].data);
178  }
179 
180  manager.initialized = 0;
181 
182  return BARELOG_NB_CORES;
183 }

185 int32_t host_mem_manager_read_mem_space(uint32_t core, barelog_event_t **events) {
186 
187 #if BARELOG_CHECK_MODE || BARELOG_DEBUG_MODE
188  if (core >= BARELOG_NB_CORES || core < 0) {
189  return BARELOG_INCONSISTENT_PARAM_ERR;
190  }
191 
192  if (events == NULL) {
193  return BARELOG_UNINITIALIZED_PARAM_ERR;
194  }
195 #endif
196 
197  int8_t ret = 0;
198  (void) ret;
199  *events = calloc(BARELOG_EVENT_PER_CORE_SHR_MEM_MAX, sizeof(barelog_event_t));
200  uint32_t n = BARELOG_EVENT_PER_CORE_SHR_MEM_MAX; // real number of events read;
201 
202  barelog_try_mutex(core);
203  barelog_set_mutex(core, 1);
204  const void *base = manager.mem_space[core].base; // On lit du cote host donc on lit dans les @virtuelles !
205  ret = manager.read(base, BARELOG_EVENT_PER_CORE_SHR_MEM_MAX*sizeof(barelog_event_t), (void *) (*events));
206 
207 #if BARELOG_CHECK_MODE || BARELOG_DEBUG_MODE
208  if (ret != BARELOG_SUCCESS) {
209  barelog_set_mutex(core, 0);
210  return BARELOG_SHRMEM_READ_ERR;
211  }
212 #endif
213  barelog_set_mutex(core, 0);
214 
215  for (uint32_t i = 0; i < BARELOG_EVENT_PER_CORE_SHR_MEM_MAX; ++i) {
216  if (strlen((*events)[i].data) == 0) { // FIXME test a ameliorer
217  n = i;
218  break;
219  }
220  }
221 
222  return n;
223 }
