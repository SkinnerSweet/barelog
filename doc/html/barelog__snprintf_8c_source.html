23  * FEATURES
24  * - careful adherence to specs regarding flags, field width and precision;
25  * - good performance for large string handling (large format, large
26  * argument or large paddings). Performance is similar to system's sprintf
27  * and in several cases significantly better (make sure you compile with
28  * optimizations turned on, tell the compiler the code is strict ANSI
29  * if necessary to give it more freedom for optimizations);
30  * - return value semantics per ISO/IEC 9899:1999 ("ISO C99");
31  * - written in standard ISO/ANSI C - requires an ANSI C compiler.
32  *
33  * SUPPORTED CONVERSION SPECIFIERS AND DATA TYPES
34  *
35  * This snprintf only supports the following conversion specifiers:
36  * s, c, d, u, o, x, X, p (and synonyms: i, D, U, O - see below)
37  * with flags: '-', '+', ' ', '0' and '#'.
38  * An asterisk is supported for field width as well as precision.
39  *
40  * Length modifiers 'h' (short int), 'l' (long int),
41  * and 'll' (long long int) are supported.
42  * NOTE:
43  * If macro SNPRINTF_LONGLONG_SUPPORT is not defined (default) the
44  * length modifier 'll' is recognized but treated the same as 'l',
45  * which may cause argument value truncation! Defining
46  * SNPRINTF_LONGLONG_SUPPORT requires that your system's sprintf also
47  * handles length modifier 'll'. long long int is a language extension
48  * which may not be portable.
49  *
50  * Conversion of numeric data (conversion specifiers d, u, o, x, X, p)
51  * with length modifiers (none or h, l, ll) is left to the system routine
52  * sprintf, but all handling of flags, field width and precision as well as
53  * c and s conversions is done very carefully by this portable routine.
54  * If a string precision (truncation) is specified (e.g. %.8s) it is
55  * guaranteed the string beyond the specified precision will not be referenced.
56  *
57  * Length modifiers h, l and ll are ignored for c and s conversions (data
58  * types wint_t and wchar_t are not supported).
59  *
60  * The following common synonyms for conversion characters are supported:
61  * - i is a synonym for d
62  * - D is a synonym for ld, explicit length modifiers are ignored
63  * - U is a synonym for lu, explicit length modifiers are ignored
64  * - O is a synonym for lo, explicit length modifiers are ignored
65  * The D, O and U conversion characters are nonstandard, they are supported
66  * for backward compatibility only, and should not be used for new code.
67  *
68  * The following is specifically NOT supported:
69  * - flag ' (thousands' grouping character) is recognized but ignored
70  * - numeric conversion specifiers: f, e, E, g, G and synonym F,
71  * as well as the new a and A conversion specifiers
72  * - length modifier 'L' (long double) and 'q' (quad - use 'll' instead)
73  * - wide character/string conversions: lc, ls, and nonstandard
74  * synonyms C and S
75  * - writeback of converted string length: conversion character n
76  * - the n$ specification for direct reference to n-th argument
77  * - locales
78  *
79  * It is permitted for str_m to be zero, and it is permitted to specify NULL
80  * pointer for resulting string argument if str_m is zero (as per ISO C99).
81  *
82  * The return value is the number of characters which would be generated
83  * for the given input, excluding the trailing null. If this value
84  * is greater or equal to str_m, not all characters from the result
85  * have been stored in str, output bytes beyond the (str_m-1) -th character
86  * are discarded. If str_m is greater than zero it is guaranteed
87  * the resulting string will be null-terminated.
88  *
89  * NOTE that this matches the ISO C99, OpenBSD, and GNU C library 2.1,
90  * but is different from some older and vendor implementations,
91  * and is also different from XPG, XSH5, SUSv2 specifications.
92  * For historical discussion on changes in the semantics and standards
93  * of snprintf see printf(3) man page in the Linux programmers manual.
94  *
95  * Routines asprintf and vasprintf return a pointer (in the ptr argument)
96  * to a buffer sufficiently large to hold the resulting string. This pointer
97  * should be passed to free(3) to release the allocated storage when it is
98  * no longer needed. If sufficient space cannot be allocated, these functions
99  * will return -1 and set ptr to be a NULL pointer. These two routines are a
100  * GNU C library extensions (glibc).
101  *
102  * Routines asnprintf and vasnprintf are similar to asprintf and vasprintf,
103  * yet, like snprintf and vsnprintf counterparts, will write at most str_m-1
104  * characters into the allocated output string, the last character in the
105  * allocated buffer then gets the terminating null. If the formatted string
106  * length (the return value) is greater than or equal to the str_m argument,
107  * the resulting string was truncated and some of the formatted characters
108  * were discarded. These routines present a handy way to limit the amount
109  * of allocated memory to some sane value.
110  *

112  * http://www.ijs.si/software/snprintf/
113  *

330 /* For copying strings longer or equal to 'breakeven_point'
331  * it is more efficient to call memcpy() than to do it inline.
332  * The value depends mostly on the processor architecture,
333  * but also on the compiler and its optimization capabilities.
334  * The value is not critical, some small value greater than zero
335  * will be just fine if you don't care to squeeze every drop
336  * of performance out of the code.
337  *
338  * Small values favor memcpy, large values favor inline code.
339  */
340 #if defined(__alpha__) || defined(__alpha)
341 # define breakeven_point 2 /* AXP (DEC Alpha) - gcc or cc or egcs */
342 #endif
343 #if defined(__i386__) || defined(__i386)
344 # define breakeven_point 12 /* Intel Pentium/Linux - gcc 2.96 */
345 #endif
346 #if defined(__hppa)
347 # define breakeven_point 10 /* HP-PA - gcc */
348 #endif
349 #if defined(__sparc__) || defined(__sparc)
350 # define breakeven_point 33 /* Sun Sparc 5 - gcc 2.8.1 */
351 #endif
352 
353 /* some other values of possible interest: */
354 /* #define breakeven_point 8 */ /* VAX 4000 - vaxc */
355 /* #define breakeven_point 19 */ /* VAX 4000 - gcc 2.7.0 */
356 
357 #ifndef breakeven_point
358 # define breakeven_point 6 /* some reasonable one-size-fits-all value */
359 #endif

361 #define fast_memcpy(d,s,n) \
362  { register size_t nn = (size_t)(n); \
363  if (nn >= breakeven_point) memcpy((d), (s), nn); \
364  else if (nn > 0) { /* proc call overhead is worth only for large strings*/\
365  register char *dd; register const char *ss; \
366  for (ss=(s), dd=(d); nn>0; nn--) *dd++ = *ss++; } }
367 
368 #define fast_memset(d,c,n) \
369  { register size_t nn = (size_t)(n); \
370  if (nn >= breakeven_point) memset((d), (int)(c), nn); \
371  else if (nn > 0) { /* proc call overhead is worth only for large strings*/\
372  register char *dd; register const int cc=(int)(c); \
373  for (dd=(d); nn>0; nn--) *dd++ = cc; } }

377 int portable_snprintf(char *str, size_t str_m, const char *fmt, /*args*/ ...) {
378  va_list ap;
379  int str_l;
380 
381  va_start(ap, fmt);
382  str_l = portable_vsnprintf(str, str_m, fmt, ap);
383  va_end(ap);
384  return str_l;
385 }

387 int portable_vsnprintf(char *str, size_t str_m, const char *fmt, va_list ap) {
388 
389  size_t str_l = 0;
390  const char *p = fmt;
391 
392 /* In contrast with POSIX, the ISO C99 now says
393  * that str can be NULL and str_m can be 0.
394  * This is more useful than the old: if (str_m < 1) return -1; */
395 
396  if (!p) p = "";
397  while (*p) {
398  if (*p != '%') {
399  /* if (str_l < str_m) str[str_l++] = *p++; -- this would be sufficient */
400  /* but the following code achieves better performance for cases
401  * where format string is long and contains few conversions */
402  const char *q = strchr(p+1,'%');
403  size_t n = !q ? strlen(p) : (q-p);
404  if (str_l < str_m) {
405  size_t avail = str_m-str_l;
406  fast_memcpy(str+str_l, p, (n>avail?avail:n));
407  }
408  p += n; str_l += n;
409  } else {
410  const char *starting_p;
411  (void)starting_p;
412  size_t min_field_width = 0, precision = 0;
413  int zero_padding = 0, precision_specified = 0, justify_left = 0;
414  int alternate_form = 0, force_sign = 0;
415  int space_for_positive = 1; /* If both the ' ' and '+' flags appear,
416  the ' ' flag should be ignored. */
417  char length_modifier = '\0'; /* allowed values: \0, h, l, L */
418  char tmp[32];/* temporary buffer for simple numeric->string conversion */
419 
420  const char *str_arg; /* string address in case of string argument */
421  size_t str_arg_l; /* natural field width of arg without padding
422  and sign */
423  unsigned char uchar_arg;
424  /* unsigned char argument value - only defined for c conversion.
425  N.B. standard explicitly states the char argument for
426  the c conversion is unsigned */
427 
428  size_t number_of_zeros_to_pad = 0;
429  /* number of zeros to be inserted for numeric conversions
430  as required by the precision or minimal field width */
431 
432  size_t zero_padding_insertion_ind = 0;
433  /* index into tmp where zero padding is to be inserted */
434 
435  char fmt_spec = '\0';
436  /* current conversion specifier character */
437 
438  str_arg = NULL;
439  starting_p = p; p++; /* skip '%' */
440  /* parse flags */
441  while (*p == '0' || *p == '-' || *p == '+' ||
442  *p == ' ' || *p == '#' || *p == '\'') {
443  switch (*p) {
444  case '0': zero_padding = 1; break;
445  case '-': justify_left = 1; break;
446  case '+': force_sign = 1; space_for_positive = 0; break;
447  case ' ': force_sign = 1;
448  /* If both the ' ' and '+' flags appear, the ' ' flag should be ignored */
449 #ifdef PERL_COMPATIBLE
450  /* ... but in Perl the last of ' ' and '+' applies */
451  space_for_positive = 1;
452 #endif
453  break;
454  case '#': alternate_form = 1; break;
455  case '\'': break;
456  }
457  p++;
458  }
459  /* If the '0' and '-' flags both appear, the '0' flag should be ignored. */
460 
461  /* parse field width */
462  if (*p == '*') {
463  int j;
464  p++; j = va_arg(ap, int);
465  if (j >= 0) min_field_width = j;
466  else { min_field_width = -j; justify_left = 1; }
467  } else if (isdigit((int)(*p))) {
468  /* size_t could be wider than unsigned int;
469  make sure we treat argument like common implementations do */
470  unsigned int uj = *p++ - '0';
471  while (isdigit((int)(*p))) uj = 10*uj + (unsigned int)(*p++ - '0');
472  min_field_width = uj;
473  }
474  /* parse precision */
475  if (*p == '.') {
476  p++; precision_specified = 1;
477  if (*p == '*') {
478  int j = va_arg(ap, int);
479  p++;
480  if (j >= 0) precision = j;
481  else {
482  precision_specified = 0; precision = 0;
483  /* NOTE:
484  * Solaris 2.6 man page claims that in this case the precision
485  * should be set to 0. Digital Unix 4.0, HPUX 10 and BSD man page
486  * claim that this case should be treated as unspecified precision,
487  * which is what we do here.
488  */
489  }
490  } else if (isdigit((int)(*p))) {
491  /* size_t could be wider than unsigned int;
492  make sure we treat argument like common implementations do */
493  unsigned int uj = *p++ - '0';
494  while (isdigit((int)(*p))) uj = 10*uj + (unsigned int)(*p++ - '0');
495  precision = uj;
496  }
497  }
498  /* parse 'h', 'l' and 'll' length modifiers */
499  if (*p == 'h' || *p == 'l') {
500  length_modifier = *p; p++;
501  if (length_modifier == 'l' && *p == 'l') { /* double l = long long */
502 #ifdef SNPRINTF_LONGLONG_SUPPORT
503  length_modifier = '2'; /* double l encoded as '2' */
504 #else
505  length_modifier = 'l'; /* treat it as a single 'l' */
506 #endif
507  p++;
508  }
509  }
510  fmt_spec = *p;
511  /* common synonyms: */
512  switch (fmt_spec) {
513  case 'i': fmt_spec = 'd'; break;
514  case 'D': fmt_spec = 'd'; length_modifier = 'l'; break;
515  case 'U': fmt_spec = 'u'; length_modifier = 'l'; break;
516  case 'O': fmt_spec = 'o'; length_modifier = 'l'; break;
517  default: break;
518  }
519  /* get parameter value, do initial processing */
520  switch (fmt_spec) {
521  case '%': /* % behaves similar to 's' regarding flags and field widths */
522  case 'c': /* c behaves similar to 's' regarding flags and field widths */
523  case 's':
524  length_modifier = '\0'; /* wint_t and wchar_t not supported */
525  /* the result of zero padding flag with non-numeric conversion specifier*/
526  /* is undefined. Solaris and HPUX 10 does zero padding in this case, */
527  /* Digital Unix and Linux does not. */
528 #if !defined(SOLARIS_COMPATIBLE) && !defined(HPUX_COMPATIBLE)
529  zero_padding = 0; /* turn zero padding off for string conversions */
530 #endif
531  str_arg_l = 1;
532  switch (fmt_spec) {
533  case '%':
534  str_arg = p; break;
535  case 'c': {
536  int j = va_arg(ap, int);
537  uchar_arg = (unsigned char) j; /* standard demands unsigned char */
538  str_arg = (const char *) &uchar_arg;
539  break;
540  }
541  case 's':
542  str_arg = va_arg(ap, const char *);
543  if (!str_arg) str_arg_l = 0;
544  /* make sure not to address string beyond the specified precision !!! */
545  else if (!precision_specified) str_arg_l = strlen(str_arg);
546  /* truncate string if necessary as requested by precision */
547  else if (precision == 0) str_arg_l = 0;
548  else {
549  /* memchr on HP does not like n > 2^31 !!! */
550  const char *q = memchr(str_arg, '\0',
551  precision <= 0x7fffffff ? precision : 0x7fffffff);
552  str_arg_l = !q ? precision : (q-str_arg);
553  }
554  break;
555  default: break;
556  }
557  break;
558  case 'd': case 'u': case 'o': case 'x': case 'X': case 'p': {
559  /* NOTE: the u, o, x, X and p conversion specifiers imply
560  the value is unsigned; d implies a signed value */
561 
562  int arg_sign = 0;
563  /* 0 if numeric argument is zero (or if pointer is NULL for 'p'),
564  +1 if greater than zero (or nonzero for unsigned arguments),
565  -1 if negative (unsigned argument is never negative) */
566 
567  int int_arg = 0; unsigned int uint_arg = 0;
568  /* only defined for length modifier h, or for no length modifiers */
569 
570  long int long_arg = 0; unsigned long int ulong_arg = 0;
571  /* only defined for length modifier l */
572 
573  void *ptr_arg = NULL;
574  /* pointer argument value -only defined for p conversion */
575 
576 #ifdef SNPRINTF_LONGLONG_SUPPORT
577  long long int long_long_arg = 0;
578  unsigned long long int ulong_long_arg = 0;
579  /* only defined for length modifier ll */
580 #endif
581  if (fmt_spec == 'p') {
582  /* HPUX 10: An l, h, ll or L before any other conversion character
583  * (other than d, i, u, o, x, or X) is ignored.
584  * Digital Unix:
585  * not specified, but seems to behave as HPUX does.
586  * Solaris: If an h, l, or L appears before any other conversion
587  * specifier (other than d, i, u, o, x, or X), the behavior
588  * is undefined. (Actually %hp converts only 16-bits of address
589  * and %llp treats address as 64-bit data which is incompatible
590  * with (void *) argument on a 32-bit system).
591  */
592 #ifdef SOLARIS_COMPATIBLE
593 # ifdef SOLARIS_BUG_COMPATIBLE
594  /* keep length modifiers even if it represents 'll' */
595 # else
596  if (length_modifier == '2') length_modifier = '\0';
597 # endif
598 #else
599  length_modifier = '\0';
600 #endif
601  ptr_arg = va_arg(ap, void *);
602  if (ptr_arg != NULL) arg_sign = 1;
603  } else if (fmt_spec == 'd') { /* signed */
604  switch (length_modifier) {
605  case '\0':
606  case 'h':
607  /* It is non-portable to specify a second argument of char or short
608  * to va_arg, because arguments seen by the called function
609  * are not char or short. C converts char and short arguments
610  * to int before passing them to a function.
611  */
612  int_arg = va_arg(ap, int);
613  if (int_arg > 0) arg_sign = 1;
614  else if (int_arg < 0) arg_sign = -1;
615  break;
616  case 'l':
617  long_arg = va_arg(ap, long int);
618  if (long_arg > 0) arg_sign = 1;
619  else if (long_arg < 0) arg_sign = -1;
620  break;
621 #ifdef SNPRINTF_LONGLONG_SUPPORT
622  case '2':
623  long_long_arg = va_arg(ap, long long int);
624  if (long_long_arg > 0) arg_sign = 1;
625  else if (long_long_arg < 0) arg_sign = -1;
626  break;
627 #endif
628  }
629  } else { /* unsigned */
630  switch (length_modifier) {
631  case '\0':
632  case 'h':
633  uint_arg = va_arg(ap, unsigned int);
634  if (uint_arg) arg_sign = 1;
635  break;
636  case 'l':
637  ulong_arg = va_arg(ap, unsigned long int);
638  if (ulong_arg) arg_sign = 1;
639  break;
640 #ifdef SNPRINTF_LONGLONG_SUPPORT
641  case '2':
642  ulong_long_arg = va_arg(ap, unsigned long long int);
643  if (ulong_long_arg) arg_sign = 1;
644  break;
645 #endif
646  }
647  }
648  str_arg = tmp; str_arg_l = 0;
649  /* NOTE:
650  * For d, i, u, o, x, and X conversions, if precision is specified,
651  * the '0' flag should be ignored. This is so with Solaris 2.6,
652  * Digital UNIX 4.0, HPUX 10, Linux, FreeBSD, NetBSD; but not with Perl.
653  */
654 #ifndef PERL_COMPATIBLE
655  if (precision_specified) zero_padding = 0;
656 #endif
657  if (fmt_spec == 'd') {
658  if (force_sign && arg_sign >= 0)
659  tmp[str_arg_l++] = space_for_positive ? ' ' : '+';
660  /* leave negative numbers for sprintf to handle,
661  to avoid handling tricky cases like (short int)(-32768) */
662 #ifdef LINUX_COMPATIBLE
663  } else if (fmt_spec == 'p' && force_sign && arg_sign > 0) {
664  tmp[str_arg_l++] = space_for_positive ? ' ' : '+';
665 #endif
666  } else if (alternate_form) {
667  if (arg_sign != 0 && (fmt_spec == 'x' || fmt_spec == 'X') )
668  { tmp[str_arg_l++] = '0'; tmp[str_arg_l++] = fmt_spec; }
669  /* alternate form should have no effect for p conversion, but ... */
670 #ifdef HPUX_COMPATIBLE
671  else if (fmt_spec == 'p'
672  /* HPUX 10: for an alternate form of p conversion,
673  * a nonzero result is prefixed by 0x. */
674 #ifndef HPUX_BUG_COMPATIBLE
675  /* Actually it uses 0x prefix even for a zero value. */
676  && arg_sign != 0
677 #endif
678  ) { tmp[str_arg_l++] = '0'; tmp[str_arg_l++] = 'x'; }
679 #endif
680  }
681  zero_padding_insertion_ind = str_arg_l;
682  if (!precision_specified) precision = 1; /* default precision is 1 */
683  if (precision == 0 && arg_sign == 0
684 #if defined(HPUX_BUG_COMPATIBLE) || defined(LINUX_COMPATIBLE)
685  && fmt_spec != 'p'
686  /* HPUX 10 man page claims: With conversion character p the result of
687  * converting a zero value with a precision of zero is a null string.
688  * Actually HP returns all zeroes, and Linux returns "(nil)". */
689 #endif
690  ) {
691  /* converted to null string */
692  /* When zero value is formatted with an explicit precision 0,
693  the resulting formatted string is empty (d, i, u, o, x, X, p). */
694  } else {
695  char f[5]; int f_l = 0;
696  f[f_l++] = '%'; /* construct a simple format string for sprintf */
697  if (!length_modifier) { }
698  else if (length_modifier=='2') { f[f_l++] = 'l'; f[f_l++] = 'l'; }
699  else f[f_l++] = length_modifier;
700  f[f_l++] = fmt_spec; f[f_l++] = '\0';
701  if (fmt_spec == 'p') str_arg_l += sprintf(tmp+str_arg_l, f, ptr_arg);
702  else if (fmt_spec == 'd') { /* signed */
703  switch (length_modifier) {
704  case '\0':
705  case 'h': str_arg_l+=sprintf(tmp+str_arg_l, f, int_arg); break;
706  case 'l': str_arg_l+=sprintf(tmp+str_arg_l, f, long_arg); break;
707 #ifdef SNPRINTF_LONGLONG_SUPPORT
708  case '2': str_arg_l+=sprintf(tmp+str_arg_l,f,long_long_arg); break;
709 #endif
710  }
711  } else { /* unsigned */
712  switch (length_modifier) {
713  case '\0':
714  case 'h': str_arg_l+=sprintf(tmp+str_arg_l, f, uint_arg); break;
715  case 'l': str_arg_l+=sprintf(tmp+str_arg_l, f, ulong_arg); break;
716 #ifdef SNPRINTF_LONGLONG_SUPPORT
717  case '2': str_arg_l+=sprintf(tmp+str_arg_l,f,ulong_long_arg);break;
718 #endif
719  }
720  }
721  /* include the optional minus sign and possible "0x"
722  in the region before the zero padding insertion point */
723  if (zero_padding_insertion_ind < str_arg_l &&
724  tmp[zero_padding_insertion_ind] == '-') {
725  zero_padding_insertion_ind++;
726  }
727  if (zero_padding_insertion_ind+1 < str_arg_l &&
728  tmp[zero_padding_insertion_ind] == '0' &&
729  (tmp[zero_padding_insertion_ind+1] == 'x' ||
730  tmp[zero_padding_insertion_ind+1] == 'X') ) {
731  zero_padding_insertion_ind += 2;
732  }
733  }
734  { size_t num_of_digits = str_arg_l - zero_padding_insertion_ind;
735  if (alternate_form && fmt_spec == 'o'
736 #ifdef HPUX_COMPATIBLE /* ("%#.o",0) -> "" */
737  && (str_arg_l > 0)
738 #endif
739 #ifdef DIGITAL_UNIX_BUG_COMPATIBLE /* ("%#o",0) -> "00" */
740 #else
741  /* unless zero is already the first character */
742  && !(zero_padding_insertion_ind < str_arg_l
743  && tmp[zero_padding_insertion_ind] == '0')
744 #endif
745  ) { /* assure leading zero for alternate-form octal numbers */
746  if (!precision_specified || precision < num_of_digits+1) {
747  /* precision is increased to force the first character to be zero,
748  except if a zero value is formatted with an explicit precision
749  of zero */
750  precision = num_of_digits+1; precision_specified = 1;
751  }
752  }
753  /* zero padding to specified precision? */
754  if (num_of_digits < precision)
755  number_of_zeros_to_pad = precision - num_of_digits;
756  }
757  /* zero padding to specified minimal field width? */
758  if (!justify_left && zero_padding) {
759  int n = min_field_width - (str_arg_l+number_of_zeros_to_pad);
760  if (n > 0) number_of_zeros_to_pad += n;
761  }
762  break;
763  }
764  default: /* unrecognized conversion specifier, keep format string as-is*/
765  zero_padding = 0; /* turn zero padding off for non-numeric convers. */
766 #ifndef DIGITAL_UNIX_COMPATIBLE
767  justify_left = 1; min_field_width = 0; /* reset flags */
768 #endif
769 #if defined(PERL_COMPATIBLE) || defined(LINUX_COMPATIBLE)
770  /* keep the entire format string unchanged */
771  str_arg = starting_p; str_arg_l = p - starting_p;
772  /* well, not exactly so for Linux, which does something inbetween,
773  * and I don't feel an urge to imitate it: "%+++++hy" -> "%+y" */
774 #else
775  /* discard the unrecognized conversion, just keep *
776  * the unrecognized conversion character */
777  str_arg = p; str_arg_l = 0;
778 #endif
779  if (*p) str_arg_l++; /* include invalid conversion specifier unchanged
780  if not at end-of-string */
781  break;
782  }
783  if (*p) p++; /* step over the just processed conversion specifier */
784  /* insert padding to the left as requested by min_field_width;
785  this does not include the zero padding in case of numerical conversions*/
786  if (!justify_left) { /* left padding with blank or zero */
787  int n = min_field_width - (str_arg_l+number_of_zeros_to_pad);
788  if (n > 0) {
789  if (str_l < str_m) {
790  size_t avail = str_m-str_l;
791  fast_memset(str+str_l, (zero_padding?'0':' '), (n>avail?avail:n));
792  }
793  str_l += n;
794  }
795  }
796  /* zero padding as requested by the precision or by the minimal field width
797  * for numeric conversions required? */
798  if (number_of_zeros_to_pad <= 0) {
799  /* will not copy first part of numeric right now, *
800  * force it to be copied later in its entirety */
801  zero_padding_insertion_ind = 0;
802  } else {
803  /* insert first part of numerics (sign or '0x') before zero padding */
804  int n = zero_padding_insertion_ind;
805  if (n > 0) {
806  if (str_l < str_m) {
807  size_t avail = str_m-str_l;
808  fast_memcpy(str+str_l, str_arg, (n>avail?avail:n));
809  }
810  str_l += n;
811  }
812  /* insert zero padding as requested by the precision or min field width */
813  n = number_of_zeros_to_pad;
814  if (n > 0) {
815  if (str_l < str_m) {
816  size_t avail = str_m-str_l;
817  fast_memset(str+str_l, '0', (n>avail?avail:n));
818  }
819  str_l += n;
820  }
821  }
822  /* insert formatted string
823  * (or as-is conversion specifier for unknown conversions) */
824  { int n = str_arg_l - zero_padding_insertion_ind;
825  if (n > 0) {
826  if (str_l < str_m) {
827  size_t avail = str_m-str_l;
828  fast_memcpy(str+str_l, str_arg+zero_padding_insertion_ind,
829  (n>avail?avail:n));
830  }
831  str_l += n;
832  }
833  }
834  /* insert right padding */
835  if (justify_left) { /* right blank padding to the field width */
836  int n = min_field_width - (str_arg_l+number_of_zeros_to_pad);
837  if (n > 0) {
838  if (str_l < str_m) {
839  size_t avail = str_m-str_l;
840  fast_memset(str+str_l, ' ', (n>avail?avail:n));
841  }
842  str_l += n;
843  }
844  }
845  }
846  }
847 #if defined(NEED_SNPRINTF_ONLY)
848  va_end(ap);
849 #endif
850  if (str_m > 0) { /* make sure the string is null-terminated
851  even at the expense of overwriting the last character
852  (shouldn't happen, but just in case) */
853  str[str_l <= str_m-1 ? str_l : str_m-1] = '\0';
854  }
855  /* Return the number of characters formatted (excluding trailing null
856  * character), that is, the number of characters that would have been
857  * written to the buffer if it were large enough.
858  *
859  * The value of str_l should be returned, but str_l is of unsigned type
860  * size_t, and snprintf is int, possibly leading to an undetected
861  * integer overflow, resulting in a negative return value, which is illegal.
862  * Both XSH5 and ISO C99 (at least the draft) are silent on this issue.
863  * Should errno be set to EOVERFLOW and EOF returned in this case???
864  */
865  return (int) str_l;
