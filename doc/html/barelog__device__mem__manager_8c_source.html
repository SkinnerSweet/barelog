52 #define barelog_get_mutex(mutex) do { \
53  if (manager.read(mutex_byte_address, 1, &(mutex)) != BARELOG_SUCCESS) { \
54  return BARELOG_SHRMEM_READ_ERR; \
55  } \
56 }while(0)

58 #define barelog_set_mutex(value) do { \
59  uint8_t __set_mutex_tmp = (value); \
60  if (manager.write(mutex_byte_address, 1, &__set_mutex_tmp) != BARELOG_SUCCESS) { \
61  return BARELOG_SHRMEM_WRITE_ERR; \
62  } \
63 } while(0)

65 #define barelog_try_mutex() do { \
66 uint8_t __mutex = 1; \
67 uint32_t __timeout = BARELOG_MUTEX_TRY_MAX; \
68 while (__mutex) { \
69  --__timeout; \
70  barelog_get_mutex(__mutex); \
71  if (__timeout <= 0) { \
72  return BARELOG_TIMEOUT_ERR; \
73  } \
74 } \
75 } while (0)

97 static inline uint32_t mod(int32_t a, int32_t b) {
98  const int32_t tmp = (a) % (b);
99 
100  uint32_t result =
101  (tmp >= 0) ? ((uint32_t) tmp) : ((uint32_t) (tmp + b));
102 
103  return result;
104 }

106 int8_t device_mem_manager_init(const uint32_t my_core,
107  const barelog_platform_t platform, const barelog_policy_t buffer_policy,
108  const barelog_policy_t memory_policy,
109  int8_t (*read)(const void * address, size_t size, void *buffer),
110  int8_t (*write)(void * address, size_t size, const void *buffer)) {
111 
112 #if BARELOG_DEBUG_MODE
113  debug_address = platform.mem_space.phy_base + BARELOG_DEBUG_OFF;
114 #endif
115 
116 #if BARELOG_CHECK_MODE || BARELOG_DEBUG_MODE
117  int8_t ret = 0;
118  if (!read || !write || !platform.mem_space.phy_base) {
119  ret = BARELOG_UNINITIALIZED_PARAM_ERR;
120  BARELOG_DEBUG(__FILE__, __LINE__, ret,
121  "device_mem_manager_init param");
122  return ret;
123  }
124 #endif
125 
126  core = my_core;
127 
128  manager.read = read;
129  manager.write = write;
130  manager.buffer_policy = buffer_policy;
131  manager.memory_policy = memory_policy;
132 
133  void *base = platform.mem_space.phy_base
134  + BARELOG_SHARED_MEM_DATA_OFFSET;
135  mem_space.phy_base = base + core * BARELOG_SHARED_MEM_PER_CORE_MAX;
136  mem_space.length = BARELOG_SHARED_MEM_PER_CORE_MAX;
137  mem_space.alignment = platform.mem_space.alignment;
138  mem_space.word_size = platform.mem_space.word_size;
139  mem_space.data = 0;
140  mem_space.base = mem_space.phy_base;
141 
142  shr_events.events = (barelog_event_t *) (mem_space.phy_base);
143  shr_events.imax = BARELOG_SHARED_MEM_PER_CORE_MAX
144  / sizeof(barelog_event_t);
145  shr_events.index = 0;
146 
147  events.head = 0;
148  events.tail = 0;
149  events.full = 0;
150  events.empty = 1;
151 
152 #if BARELOG_SAFE_MODE
153  mutex_byte_address = platform.mem_space.phy_base + core;
154 #endif
155 
156  return BARELOG_NB_CORES;
157 }

159 int8_t device_mem_manager_write_buffer(barelog_event_t event) {
160  if (events.full) {
161  int8_t ret = 0;
162  (void) ret;
163  switch (manager.buffer_policy) {
164  case SKIP:
165  return BARELOG_SUCCESS;
166  break;
167  case REPLACE:
168  events.tail = (events.tail + 1)
169  % BARELOG_EVENT_PER_CORE_MAX;
170  break;
171  case FLUSH:
172  ret = device_mem_manager_flush_buffer();
173 #if BARELOG_CHECK_MODE || BARELOG_DEBUG_MODE
174  if (ret != BARELOG_SUCCESS) {
175  BARELOG_DEBUG(__FILE__, __LINE__, ret,
176  "device_mem_manager_flush_buffer call");
177  return ret;
178  }
179 #endif
180  break;
181  case DESTROY:
182  ret = device_mem_manager_flush_buffer();
183 #if BARELOG_CHECK_MODE || BARELOG_DEBUG_MODE
184  if (ret != BARELOG_SUCCESS) {
185  BARELOG_DEBUG(__FILE__, __LINE__, ret,
186  "device_mem_manager_flush_buffer call");
187  return ret;
188  }
189 #endif
190  ret = device_mem_manager_clean_buffer();
191 #if BARELOG_CHECK_MODE || BARELOG_DEBUG_MODE
192  if (ret != BARELOG_SUCCESS) {
193  BARELOG_DEBUG(__FILE__, __LINE__, ret,
194  "device_mem_manager_cleah_buffer call");
195  return ret;
196  }
197 #endif
198  break;
199  default:
200  BARELOG_DEBUG(__FILE__, __LINE__, BARELOG_ERR, "unrecognized policy");
201  return BARELOG_ERR;
202  }
203  }
204 
205  events.buffer[events.head] = event;
206  events.buffer[events.head].core = core;
207  events.empty = 0;
208 
209  events.head = (events.head + 1) % BARELOG_EVENT_PER_CORE_MAX;
210 
211  /* If the next case to fulfill is already taken */
212  if (events.head == events.tail) {
213  /* The buffer is full */
214  events.full = 1;
215  }
216 
217  return BARELOG_SUCCESS;
218 }

220 inline int8_t device_mem_manager_clean_buffer(void) {
221  uint32_t events_to_read =
222  (events.full) ?
223  BARELOG_EVENT_PER_CORE_MAX :
224  (mod((events.head - events.tail),
225  BARELOG_EVENT_PER_CORE_MAX));
226  return device_mem_manager_clean(events_to_read);
227 }

229 int8_t device_mem_manager_clean(uint32_t n) {
230 #if BARELOG_CHECK_MODE || BARELOG_DEBUG_MODE
231  int8_t ret = 0;
232  if (n <= 0 || n > BARELOG_EVENT_PER_CORE_MAX) {
233  ret = BARELOG_INCONSISTENT_PARAM_ERR;
234  BARELOG_DEBUG(__FILE__, __LINE__, ret,
235  "device_mem_manager_clean param");
236  return ret;
237  }
238 #endif
239 
240  if (events.empty) {
241  return BARELOG_SUCCESS;
242  }
243 
244  uint32_t ind = 0;
245  uint32_t imax = mod((events.head - 1), BARELOG_EVENT_PER_CORE_MAX);
246  for (uint32_t i = 0; i < n; ++i) {
247  ind = (events.tail + i) % BARELOG_EVENT_PER_CORE_MAX;
248  events.buffer[ind] = BARELOG_EVENT_INITIALIZER;
249  if (ind == imax) {
250  events.empty = 1;
251  break;
252  }
253  }
254 
255  events.tail = (ind + 1) % BARELOG_EVENT_PER_CORE_MAX;
256  events.full = 0;
257 
258  return BARELOG_SUCCESS;
259 }

270 int8_t device_mem_manager_flush(uint32_t n) {
271  int8_t ret = 0;
272  (void) ret;
273 
274 #if BARELOG_CHECK_MODE || BARELOG_DEBUG_MODE
275  if (n <= 0 || n > BARELOG_EVENT_PER_CORE_MAX) {
276  ret = BARELOG_INCONSISTENT_PARAM_ERR;
277  BARELOG_DEBUG(__FILE__, __LINE__, ret,
278  "device_mem_manager_clean param");
279  return ret;
280  }
281 #endif
282 
283  uint32_t nmax = n;
284  uint32_t n1 = 0;
285  uint32_t n2 = 0;
286 
287  uint32_t events_to_read =
288  (events.full) ?
289  BARELOG_EVENT_PER_CORE_MAX :
290  (mod((events.head - events.tail),
291  BARELOG_EVENT_PER_CORE_MAX));
292 
293  if (events_to_read == 0) {
294  return BARELOG_SUCCESS;
295  }
296 
297  // We make sure we don't read more events than there are available.
298  nmax = (n > events_to_read) ? events_to_read : n;
299 
300  uint32_t barelog_total_events_size = nmax * sizeof(barelog_event_t);
301 
302  if ((shr_events.imax - shr_events.index + 1) < nmax) {
303  switch (manager.memory_policy) {
304  case SKIP:
305  return BARELOG_SUCCESS;
306  break;
307  case REPLACE:
308  shr_events.index = 0;
309  break;
310  case DESTROY:
311  shr_events.index = 0;
312  ret = device_mem_manager_clean_memory();
313 #if BARELOG_CHECK_MODE || BARELOG_DEBUG_MODE
314  if (ret != BARELOG_SUCCESS) {
315  BARELOG_DEBUG(__FILE__, __LINE__, ret,
316  "device_mem_manager_clean_memory call");
317  return ret;
318  }
319 #endif
320  break;
321  default:
322  BARELOG_DEBUG(__FILE__, __LINE__, BARELOG_ERR, "unrecognized policy");
323  return BARELOG_ERR;
324  }
325  }
326 
327  barelog_try_mutex(); barelog_set_mutex(1);
328 
329  /* If there are too many events, we must have to separate the writings
330  * in two : first the section from tail to BARELOG_EVENT_PER_CORE_MAX - 1
331  * then the section from 0 to nmax.
332  *
333  * Otherwise we just flush the buffer from the tail of the queue to nmax.
334  */
335  if ((events.tail + nmax - 1) >= BARELOG_EVENT_PER_CORE_MAX) {
336  n1 = BARELOG_EVENT_PER_CORE_MAX - events.tail;
337  if (n1) {
338  if (manager.write(
339  &(shr_events.events[shr_events.index]),
340  n1 * sizeof(barelog_event_t),
341  (const void *) (&(events.buffer[events.tail]))) != BARELOG_SUCCESS) {
342  barelog_set_mutex(0);
343  ret = BARELOG_SHRMEM_WRITE_ERR;
344  BARELOG_DEBUG(__FILE__, __LINE__, ret,
345  "shared memory writing error");
346  return ret;
347  }
348  }
349  n2 = nmax - n1;
350  if (n2) {
351  if (manager.write(
352  &(shr_events.events[shr_events.index]),
353  n2 * sizeof(barelog_event_t),
354  (const void *) (&(events.buffer[0]))) != BARELOG_SUCCESS) {
355  barelog_set_mutex(0);
356  ret = BARELOG_SHRMEM_WRITE_ERR;
357  BARELOG_DEBUG(__FILE__, __LINE__, ret,
358  "shared memory writing error");
359  return ret;
360  }
361  }
362  } else {
363  if (manager.write(&(shr_events.events[shr_events.index]),
364  barelog_total_events_size,
365  (const void *) (&(events.buffer[events.tail]))) != BARELOG_SUCCESS) {
366  barelog_set_mutex(0);
367 
368  ret = BARELOG_SHRMEM_WRITE_ERR;
369  BARELOG_DEBUG(__FILE__, __LINE__, ret,
370  "shared memory writing error");
371  return ret;
372  }
373  } barelog_set_mutex(0);
374 
375  shr_events.index += nmax;
376 
377  ret = BARELOG_SUCCESS;
378  BARELOG_DEBUG(__FILE__, __LINE__, ret, "flushing success !");
379 
380  return BARELOG_SUCCESS;
381 }

383 int8_t device_mem_manager_clean_memory(void) {
384  barelog_try_mutex(); barelog_set_mutex(1);
385  memset(mem_space.base, 0, mem_space.length);
386  barelog_set_mutex(0);
387 
388  shr_events.index = 0;
389 
390  return BARELOG_SUCCESS;
391 }
