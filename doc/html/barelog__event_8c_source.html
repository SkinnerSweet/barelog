39 int8_t barelog_event_to_string(const barelog_event_t event, char *buffer) {
40 #if BARELOG_CHECK_MODE
41  if (strlen(event.data) > EVENT_TO_STRING_SIZE) {
42  return BARELOG_INCONSISTENT_PARAM_ERR;
43  }
44 #endif
45 
46  return snprintf(buffer, EVENT_TO_STRING_SIZE, "%"PRIu32" %"PRIu32" %s",
47  event.timestamp, event.core, event.data);
48 }

51 int8_t barelog_events_to_strings(const barelog_event_t *events, size_t n, barelog_result_buffer_t *res) {
52 #if BARELOG_CHECK_MODE
53  if (n < 0) {
54  return BARELOG_INCONSISTENT_PARAM_ERR;
55  }
56  if (!res) {
57  return BARELOG_UNINITIALIZED_PARAM_ERR;
58  }
59 #endif
60 
61  res->buffer = calloc(n, sizeof(char *));
62 
63 #if BARELOG_CHECK_MODE
64  if (res->buffer == NULL) {
65  return BARELOG_ERR;
66  }
67 #endif
68  res->buffer_length = n;
69  res->sub_buffer_length = EVENT_TO_STRING_SIZE;
70 
71  int8_t ret = 0;
72  (void) ret;
73  for (uint32_t i = 0; i < n; ++i) {
74  res->buffer[i] = (char *)calloc(EVENT_TO_STRING_SIZE, sizeof(char));
75  ret = barelog_event_to_string(events[i], res->buffer[i]);
76 #if BARELOG_CHECK_MODE
77  if (ret != BARELOG_SUCCESS) {
78  return BARELOG_EVENT_CONVERSION_ERR;
79  }
80 #endif
81  }
82 
83  return ret;
84 }
