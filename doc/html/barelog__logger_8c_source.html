62 int8_t barelog_start(void) {
63  int8_t ret = 0;
64 #if BARELOG_CHECK_MODE || BARELOG_DEBUG_MODE
65  if (!logger.init_clock || !logger.start_clock) {
66  ret = BARELOG_UNINITIALIZED_PARAM_ERR;
67  BARELOG_DEBUG(__FILE__, __LINE__, ret, "barelog_start param");
68  return ret;
69  }
70 #endif
71  ret = logger.init_clock();
72  return (ret + logger.start_clock());
73 }

75 int8_t barelog_log(barelog_lvl_t lvl, const char *format, ...) {
76 
77  if (lvl > logger.log_lvl) {
78  return -1;
79  }
80 
81  barelog_event_t event = BARELOG_EVENT_INITIALIZER;
82 
83  event.timestamp = logger.get_clock();
84 
85  va_list ap;
86  va_start(ap, format);
87  portable_vsnprintf(event.data, BARELOG_BUF_MAX_SIZE, format, ap);
88  //vsnprintf(event.data, BARELOG_BUF_MAX_SIZE, format, ap);
89 
90  va_end(ap);
91 
92  return device_mem_manager_write_buffer(event);
93 
94 }

96 int8_t barelog_immediate_log(barelog_lvl_t lvl, const char *format, ...) {
97  if (lvl > logger.log_lvl) {
98  return -1;
99  }
100  int8_t ret = 0;
101  va_list ap;
102  va_start(ap, format);
103  ret += barelog_log(lvl, format, ap);
104  ret += barelog_flush(1);
105  ret += barelog_clean(1);
106  va_end(ap);
107 
108  return ret;
109 }
